Database Schema

5 tables for Phase 1. Column definitions, types, constraints, indexes, and SQL migrations.

Multi-tenancy model: Shared database with hospital_id column scoping. Every staff-facing query is automatically scoped to the user's hospital via middleware. Super admin queries have no scope restriction.

Encryption: Fields marked ENC are AES-256-GCM encrypted at the application layer before storage. The DB stores ciphertext. Decryption happens in the service layer only.

Audit log: The audit_logs table is append-only. The application DB user has NO UPDATE or DELETE privilege on this table.

Key rules for users table:
1. email_hash is SHA-256(email.toLowerCase()) — used for unique constraint and fast lookup without decryption.
2. hospital_id is NULL only for super_admin role — enforced at application layer.
3. status starts as 'invited' when created via invite flow. Changes to 'active' on first login.
4. created_by tracks who invited this user — important for audit trail.

Key rules for invites table:
1. The raw token is NEVER stored. Only SHA-256(token) is stored. The raw token is sent in the email link only.
2. When verifying: hash the incoming token, look up token_hash, check used=false AND expires_at > now().
3. After use: set used=true immediately — even before creating the user — to prevent race conditions.

Key rules for audit_logs table:
1. APPEND ONLY — the application DB user must have SELECT and INSERT only. No UPDATE or DELETE.
2. Log EVERY write action: logins, logouts, creates, updates, deletes, role changes, deactivations.
3. action strings use SCREAMING_SNAKE_CASE constants: USER_LOGIN, HOSPITAL_CREATED, STAFF_INVITED, etc.
4. old_value / new_value store JSON snapshots — strip sensitive fields (password_hash) before logging.

-- Phase 1 Migrations — Run in order
-- ================================================

-- 001_create_enums.sql
CREATE TYPE user_role AS ENUM (
  'super_admin', 'admin', 'doctor',
  'receptionist', 'lab_tech', 'pharmacist', 'nurse'
);
CREATE TYPE user_status   AS ENUM ('invited', 'active', 'deactivated');
CREATE TYPE hospital_plan AS ENUM ('basic', 'pro', 'enterprise');
CREATE TYPE hospital_status AS ENUM ('pending', 'active', 'inactive');
CREATE TYPE invite_type   AS ENUM ('invite', 'password_reset');

-- 002_create_hospitals.sql
CREATE TABLE hospitals (
  id           UUID           PRIMARY KEY DEFAULT gen_random_uuid(),
  name         TEXT           NOT NULL,
  address      TEXT           NOT NULL,  -- AES-256 encrypted
  city         TEXT           NOT NULL,
  branch_code  TEXT           NOT NULL UNIQUE,
  phone        TEXT,
  plan         hospital_plan  NOT NULL DEFAULT 'basic',
  status       hospital_status NOT NULL DEFAULT 'pending',
  created_by   UUID           NOT NULL,
  created_at   TIMESTAMPTZ    NOT NULL DEFAULT now(),
  updated_at   TIMESTAMPTZ    NOT NULL DEFAULT now()
);

-- 003_create_users.sql
CREATE TABLE users (
  id             UUID         PRIMARY KEY DEFAULT gen_random_uuid(),
  email          TEXT         NOT NULL,        -- AES-256 encrypted
  email_hash     TEXT         NOT NULL UNIQUE, -- SHA-256 for lookup
  password_hash  TEXT         NOT NULL,
  first_name     TEXT         NOT NULL,
  last_name      TEXT         NOT NULL,
  phone          TEXT,                          -- AES-256 encrypted
  role           user_role    NOT NULL,
  hospital_id    UUID         REFERENCES hospitals(id) ON DELETE RESTRICT,
  status         user_status  NOT NULL DEFAULT 'invited',
  last_login_at  TIMESTAMPTZ,
  created_by     UUID         REFERENCES users(id) ON DELETE SET NULL,
  created_at     TIMESTAMPTZ  NOT NULL DEFAULT now(),
  updated_at     TIMESTAMPTZ  NOT NULL DEFAULT now()
);
CREATE INDEX idx_users_hospital_id  ON users(hospital_id);
CREATE INDEX idx_users_email_hash   ON users(email_hash);
CREATE INDEX idx_users_role         ON users(role);

-- 004_create_invites.sql
CREATE TABLE invites (
  id           UUID          PRIMARY KEY DEFAULT gen_random_uuid(),
  email        TEXT          NOT NULL,
  type         invite_type   NOT NULL,
  role         user_role,
  hospital_id  UUID          REFERENCES hospitals(id) ON DELETE CASCADE,
  token_hash   TEXT          NOT NULL,
  expires_at   TIMESTAMPTZ   NOT NULL,
  used         BOOLEAN       NOT NULL DEFAULT false,
  created_by   UUID          REFERENCES users(id) ON DELETE SET NULL,
  created_at   TIMESTAMPTZ   NOT NULL DEFAULT now()
);
CREATE INDEX idx_invites_token_hash ON invites(token_hash);

-- 005_create_settings.sql
CREATE TABLE settings (
  id                   UUID         PRIMARY KEY DEFAULT gen_random_uuid(),
  hospital_id          UUID         NOT NULL UNIQUE REFERENCES hospitals(id) ON DELETE CASCADE,
  working_hours        JSONB,
  departments          JSONB        DEFAULT '[]',
  services             JSONB        DEFAULT '[]',
  billing_config       JSONB,
  onboarding_checklist JSONB        NOT NULL DEFAULT '{"admin_assigned":false,"doctor_added":false,"working_hours_set":false,"billing_configured":false,"reception_added":false}',
  updated_at           TIMESTAMPTZ  NOT NULL DEFAULT now()
);

-- 006_create_audit_logs.sql
CREATE TABLE audit_logs (
  id            UUID         PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id       UUID,        -- no FK — user may be deleted
  user_role     TEXT,
  hospital_id   UUID,        -- no FK — hospital may be deleted
  action        TEXT         NOT NULL,
  resource_type TEXT,
  resource_id   UUID,
  old_value     JSONB,
  new_value     JSONB,
  ip_address    INET,
  user_agent    TEXT,
  created_at    TIMESTAMPTZ  NOT NULL DEFAULT now()
);
CREATE INDEX idx_audit_user_id    ON audit_logs(user_id);
CREATE INDEX idx_audit_hospital   ON audit_logs(hospital_id);
CREATE INDEX idx_audit_action     ON audit_logs(action);
CREATE INDEX idx_audit_created_at ON audit_logs(created_at);

-- 007_updated_at_trigger.sql
CREATE OR REPLACE FUNCTION set_updated_at()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = now();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_users_updated_at
  BEFORE UPDATE ON users
  FOR EACH ROW EXECUTE FUNCTION set_updated_at();

CREATE TRIGGER trg_hospitals_updated_at
  BEFORE UPDATE ON hospitals
  FOR EACH ROW EXECUTE FUNCTION set_updated_at();
